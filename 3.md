今までなぜか map を使っていた
```cpp
class Solution {
public:
    bool isAnagram(string s, string t) {
       if (s.length() != t.length()) {
            return false;
       }
       unordered_map<char, int> count_char;
       for (int i = 0; i < s.length(); ++i) {
            ++count_char[s[i]];
            --count_char[t[i]];
       }
       for (const auto& kv: count_char) {
            if(kv.second != 0) {
                return false;
            }
       }
       return true;
    }
};
```
